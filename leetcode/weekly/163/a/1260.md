## 方法一：创建新数组

题目说，位于 $\textit{grid}[i][n - 1]$ 的元素会移动到 $\textit{grid}[i + 1][0]$，也就是行末元素的「下一个元素」是下一行的第一个元素。

那么，把二维矩阵的每一行拼在一起，可以得到一个一维数组，方便我们处理。

示例 1 的矩阵转成一维数组，得 $a = [1,2,3,4,5,6,7,8,9]$。迁移（循环右移）$k=1$ 次，得到数组 $b = [9,1,2,3,4,5,6,7,8]$。再把 $b$ 转回矩阵，得到下图右边的答案。

![lc1260.png](https://pic.leetcode.cn/1784504087-CoShaO-lc1260.png){:width=400px}

一般地，长为 $N$ 的数组 $a$ 循环右移 $k$ 次后，下标为 $i$ 的元素移动到下标 $(i+k)\bmod N$。

代码实现时，并不需要真的创建长为 $N = mn$ 的数组 $a$ 和 $b$。

1. 算出 $\textit{grid}[i][j]$ 在 $a$ 中的下标。由于每行有 $n$ 个元素，在 $\textit{grid}[i][j]$ 之前有完整的 $i$ 行，即 $in$ 个数，加上 $\textit{grid}[i][j]$ 这行的前 $j+1$ 个数，得到 $\textit{grid}[i][j]$ 是 $a$ 中的第 $in+j+1$ 个数，即 $a[in+j]$。
2. 循环右移 $k$ 次后，下标为 $in+j$ 的元素移动到下标 $p = (in+j+k)\bmod N$。
3. 找到 $p$ 在答案中的行号和列号。由于答案有 $n$ 列，所以下标 $p$ 之前有完整的 $\lfloor p/n \rfloor$ 行。下标 $p$ 在当前这一行的 $p\bmod n$ 列。

```py [sol-Python3]
class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        m, n = len(grid), len(grid[0])
        ans = [[0] * n for _ in range(m)]

        size = m * n
        for i, row in enumerate(grid):
            for j, x in enumerate(row):
                p = (i * n + j + k) % size
                ans[p // n][p % n] = x
        return ans
```

```java [sol-Java]
class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int m = grid.length;
        int n = grid[0].length;
        List<List<Integer>> ans = new ArrayList<>(m);
        for (int i = 0; i < m; i++) {
            ans.add(new ArrayList<>(Collections.nCopies(n, 0)));
        }

        int size = m * n;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int p = (i * n + j + k) % size;
                ans.get(p / n).set(p % n, grid[i][j]);
            }
        }
        return ans;
    }
}
```

## 方法二：三次反转

利用 [189. 轮转数组](https://leetcode.cn/problems/rotate-array/) 的技巧，可以做到 $\mathcal{O}(1)$ 空间。原理见 [我的题解](https://leetcode.cn/problems/rotate-array/solutions/2784427/tu-jie-yuan-di-zuo-fa-yi-tu-miao-dong-py-ryfv/)。

```py [sol-Python3]
class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        m, n = len(grid), len(grid[0])

        def reverse(l: int, r: int) -> None:
            while l < r:
                x1, y1 = divmod(l, n)
                x2, y2 = divmod(r, n)
                grid[x1][y1], grid[x2][y2] = grid[x2][y2], grid[x1][y1]
                l += 1
                r -= 1

        # 189. 轮转数组
        size = m * n
        k %= size  # 轮转 k 次等同于轮转 k % size 次
        reverse(0, size - 1)
        reverse(0, k - 1)
        reverse(k, size - 1)
        return grid
```

```java [sol-Java]
class Solution {
    // 注意：由于返回值不是 int[][]，无法原地
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        // 189. 轮转数组
        int m = grid.length;
        int n = grid[0].length;
        int size = m * n;
        k %= size; // 轮转 k 次等同于轮转 k % size 次
        reverse(grid, n, 0, size - 1);
        reverse(grid, n, 0, k - 1);
        reverse(grid, n, k, size - 1);

        List<List<Integer>> ans = new ArrayList<>(m);
        for (int[] row : grid) {
            List<Integer> r = new ArrayList<>(n);
            for (int x : row) {
                r.add(x);
            }
            ans.add(r);
        }
        return ans;
    }

    private void reverse(int[][] grid, int n, int l, int r) {
        while (l < r) {
            int x1 = l / n;
            int y1 = l % n;
            int x2 = r / n;
            int y2 = r % n;
            int tmp = grid[x1][y1];
            grid[x1][y1] = grid[x2][y2];
            grid[x2][y2] = tmp;
            l++;
            r--;
        }
    }
}
```

```cpp [sol-C++]
class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int m = grid.size(), n = grid[0].size();

        auto reverse = [&](int l, int r) -> void {
            while (l < r) {
                int x1 = l / n, y1 = l % n;
                int x2 = r / n, y2 = r % n;
                swap(grid[x1][y1], grid[x2][y2]);
                l++;
                r--;
            }
        };

        // 189. 轮转数组
        int size = m * n;
        k %= size; // 轮转 k 次等同于轮转 k % size 次
        reverse(0, size - 1);
        reverse(0, k - 1);
        reverse(k, size - 1);
        return grid;
    }
};
```

```go [sol-Go]
func shiftGrid(grid [][]int, k int) [][]int {
	m, n := len(grid), len(grid[0])

	reverse := func(l, r int) {
		for l < r {
			x1, y1 := l/n, l%n
			x2, y2 := r/n, r%n
			grid[x1][y1], grid[x2][y2] = grid[x2][y2], grid[x1][y1]
			l++
			r--
		}
	}

	// 189. 轮转数组
	size := m * n
	k %= size // 轮转 k 次等同于轮转 k % size 次
	reverse(0, size-1)
	reverse(0, k-1)
	reverse(k, size-1)
	return grid
}
```

#### 复杂度分析

- 时间复杂度：$\mathcal{O}(mn)$，其中 $m$ 和 $n$ 分别是 $\textit{grid}$ 的行数和列数。
- 空间复杂度：$\mathcal{O}(1)$。
